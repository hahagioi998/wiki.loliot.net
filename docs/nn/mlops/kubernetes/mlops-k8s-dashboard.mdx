## Installation

```shell
mkdir -p k8s-dashboard/base
```

```shell
wget https://raw.githubusercontent.com/kubernetes/dashboard/v2.4.0/aio/deploy/recommended.yaml \
    -O k8s-dashboard/base/k8s-dashboard.yaml
```

```yaml title="k8s-dashboard/base/k8s-dashboard-admin.yaml"
apiVersion: v1
kind: ServiceAccount
metadata:
  name: k8s-dashboard-admin
  namespace: kubernetes-dashboard

---
apiVersion: rbac.authorization.k8s.io/v1
kind: ClusterRoleBinding
metadata:
  name: k8s-dashboard-admin
roleRef:
  apiGroup: rbac.authorization.k8s.io
  kind: ClusterRole
  name: cluster-admin
subjects:
  - kind: ServiceAccount
    name: k8s-dashboard-admin
    namespace: kubernetes-dashboard
```

```yaml title="k8s-dashboard/base/kustomization.yaml"
apiVersion: kustomize.config.k8s.io/v1beta1
kind: Kustomization
resources:
  - k8s-dashboard.yaml
  - k8s-dashboard-admin.yaml
```

```shell
k8s-dashboard
└── base/
    ├── k8s-dashboard-admin.yaml
    ├── k8s-dashboard.yaml
    └── kustomization.yaml
```

```shell
kustomize build k8s-dashboard/base | kubectl apply -f -
```
